\begin{frame}{Takeaways}
  \begin{itemize}
    \item \textcolor{DarkPrimary}{Forecasting bug fixed:} previously the reservoir state $\vct{h}_t$ was \textcolor{LightPrimary}{not} advanced within the forecast window; it is now rolled forward step--by--step during forecasting.
    \item Ran the full pipeline on \textcolor{LightPrimary}{all three} simulation scenarios \textcolor{LightPrimary}{and} on the San Lorenzo River daily data.
    \item Computed metrics \& diagnostics for each run \textcolor{LightPrimary}{(CRPS, PinballMean, ELBO, PIT, rolling coverages, synthesized 95\% bands)}.
    \item \textcolor{DarkPrimary}{Models are not calibrated/optimized yet} (same architecture/hyperparameters across datasets).
  \end{itemize}
\end{frame}

%========================
% Q-DESN
%========================
\begin{frame}{Q-DESN}
\scriptsize
\tightmathbox{
&\textbf{Obs.:}&&
 y_t\mid(\cdot)\sim \Normal\!\Big(\mu_t+\lambda(\Lparam{\gamma})\,\Lparam{\sigma}\, \textcolor{purple}{s}_t
                                   +A(\Lparam{\gamma})\,\textcolor{purple}{v}_t,\ 
                                   B(\Lparam{\gamma})\,\Lparam{\sigma}\, \textcolor{purple}{v}_t\Big),\quad
 Q_{p_0}(y_t\mid\cdot)=\mu_t \\
&\textbf{Latents:}&&
\textcolor{purple}{s}_t\sim N^+(0,1),\quad \textcolor{purple}{v}_t\mid\Lparam{\sigma}\sim\mathrm{Exp}\!\big(1/\Lparam{\sigma}\big)\\[0.25em]
&\textbf{Features:}&&
\vct{x}_t=\big[\vct{h}_{t,D}^\top,\ k(\tilde{\vct{h}}_{t,1})^\top,\dots,k(\tilde{\vct{h}}_{t,D-1})^\top,\ \textcolor{ForestGreen}{\vct{u}_t}^{\!\top}\big]^\top,\quad
\mu_t=\vct{x}_t^\top\,\Lparam{\vct{\beta}}\\[0.25em]
&\textbf{Reservoir:}&&
\vct{h}_{t,d}=(1-\alpha)\,\vct{h}_{t-1,d}+\alpha\,\vct{\omega}_{t,d},\quad
\vct{\omega}_{t,1}=f\!\Big(\tfrac{\rho_1}{|\lambda_{\max}(\mtr{W}_1)|}\,\mtr{W}_1\,\vct{h}_{t-1,1}
                  +\mtr{W}_1^{\mathrm{in}}\,\textcolor{ForestGreen}{\vct{u}_t}\Big),\\
&&&\tilde{\vct{h}}_{t,d-1}=\mtr{Q}_{d-1}\,\vct{h}_{t,d-1},\ \ 
\vct{\omega}_{t,d}=f\!\Big(\tfrac{\rho_d}{|\lambda_{\max}(\mtr{W}_d)|}\,\mtr{W}_d\,\vct{h}_{t-1,d}
                  +\mtr{W}_d^{\mathrm{in}}\,\tilde{\vct{h}}_{t,d-1}\Big)\ (d\ge2)\\[0.25em]
}
\vspace{0.35em}

{\tiny
\textit{Input design: }\(\textcolor{ForestGreen}{\vct{u}_t}
= \big[\,y_{t-1},\dots,y_{t-m},\ \vct{z}_t^{\!\top}\big]^{\!\top}\in\mathbb{R}^{p}\), with \(p=m+r\) and \(\vct{z}_t\in\mathbb{R}^{r}\).
\par
\textit{Dims: } \(\vct{h}_{t,d}\in\mathbb{R}^{n_d}\);
\(\mtr{W}_d\in\mathbb{R}^{n_d\times n_d}\);
\(\mtr{W}_1^{\mathrm{in}}\in\mathbb{R}^{n_1\times p}\);
\(\mtr{W}_d^{\mathrm{in}}\in\mathbb{R}^{n_d\times n_{d-1}}\) for \(d\ge2\).
}
\end{frame}

\begin{frame}{Q-DESN: Priors}
\scriptsize
\begin{columns}[T,onlytextwidth]
  \column{0.54\textwidth}
    \begin{whiteblock}{\small Scale + skew}
    \[
    \begin{aligned}
      \sigma &\sim IG(a_\sigma,b_\sigma), && a_\sigma>0,\ b_\sigma>0,\\
      \gamma &\sim \pi_\gamma(\gamma)\,\mathbf{1}(L<\gamma<U),
      && \pi_\gamma \text{ on } (L,U).
    \end{aligned}
    \]
    \vspace{-0.3em}
    {\tiny $(L,U)$ is the $p$-dependent support from the quantile-fixing construction.}
    \end{whiteblock}

    \begin{whiteblock}{\small RHS for $\vct{\beta}$}
    \[
    \begin{aligned}
      V_j(\lambda_j,\tau,c^2) &= \tau^2 \frac{c^2 \lambda_j^2}{c^2+\tau^2 \lambda_j^2},\\
      \beta_j \mid \lambda_j,\tau,c^2 &\sim N(0,V_j),\quad j=1,\dots,k.
    \end{aligned}
    \]
    \end{whiteblock}

  \column{0.46\textwidth}
    \begin{whiteblock}{\small Hierarchical scales}
    \[
    \begin{aligned}
      \lambda_j &\sim C^+(0,1),\quad j=1,\dots,k,\\
      \tau &\sim C^+(0,\tau_0),\quad
      c^2 \sim IG(\nu/2,\nu s^2/2).
    \end{aligned}
    \]
    \vspace{-0.2em}
    {\tiny $\lambda_j$ local, $\tau$ global, $c^2$ slab; $\tau_0,\nu,s$ tune sparsity and slab width.}
    \end{whiteblock}
\end{columns}
\end{frame}
