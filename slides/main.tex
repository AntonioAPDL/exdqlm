\newcommand{\ph}[1]{\textcolor{gray}{\footnotesize[\,#1\,]}}
\newcommand{\phplot}[2][0.28\textheight]{%
  \fbox{\begin{minipage}[c][#1][c]{0.95\linewidth}\centering
  \footnotesize\textit{#2}
  \end{minipage}}}
\newcommand{\phwide}[1]{%
  \fbox{\begin{minipage}[c][0.36\textheight][c]{0.92\linewidth}\centering
  \footnotesize\textit{#1}
  \end{minipage}}}

\newenvironment{whiteblock}[1]{%
  \begin{block}{#1}%
}{\end{block}}

% Section divider frame
\newcommand{\sectionframe}[1]{%
  \section{#1}
  \begin{frame}[plain]
    \vfill
    {\Huge\bfseries\color{DarkPrimary} #1\par}
    \vspace{0.6em}
    {\large\color{gray} \ph{insert 1-line purpose for this section}}
    \vfill
  \end{frame}
}
