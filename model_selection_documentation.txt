\section{Purpose and scope}
This document specifies a rigorous, time-series-safe model selection workflow for QDESN. Multiple quantile models (indexed by $\tau$) share a common DESN specification and are synthesized into a single predictive distribution. The goal is to select DESN hyperparameters efficiently and reproducibly using out-of-sample predictive performance of the \emph{synthesized} distribution (primary) plus calibration of the \emph{individual} quantile models (secondary/constraint or penalty).

\paragraph{Selection target.}
We select the shared DESN specification while holding secondary parameters fixed. In repo terms, the selection target is the \texttt{cfg$desn} block (\texttt{D, n, n\_tilde, m, alpha, rho, pi\_w, pi\_in, washout, add\_bias, seed}) with readout and VB settings fixed unless explicitly varied.

\paragraph{Key design principles.}
\begin{enumerate}[label=(\arabic*), itemsep=2pt, topsep=4pt]
\item Strict time-series evaluation (no leakage): train/verification/test split and rolling-origin evaluation on the verification block.
\item Proper scoring rules for distributional forecasts: CRPS for the synthesized predictive distribution.
\item Calibration defined by \textbf{CalCRPS only} (uncertainty-aware; see \S\ref{sec:calcrps}).
\item Explicit handling of ESN stochasticity: common random numbers across candidates; average objectives across seeds.
\item Multi-stage, multi-fidelity search: broad screening, then racing/successive halving, then final robust evaluation.
\item Caching and parallelization: reuse reservoir states per (candidate, seed) across quantiles; parallelize across candidates/seeds/quantiles.
\end{enumerate}

\section{Model components and notation}

\subsection{Time series and splits}
Let $y_{1:T}$ denote the observed time series. We use a time-ordered split:
\[
\{1,\dots,T\} = \mathcal{T}_{\text{train}} \cup \mathcal{T}_{\text{verify}} \cup \mathcal{T}_{\text{test}},
\quad \max(\mathcal{T}_{\text{train}}) < \min(\mathcal{T}_{\text{verify}}) < \min(\mathcal{T}_{\text{test}}).
\]
\begin{itemize}[itemsep=2pt, topsep=4pt]
\item \textbf{Training block} $\mathcal{T}_{\text{train}}$: used for fitting each quantile model under a candidate DESN.
\item \textbf{Verification block} $\mathcal{T}_{\text{verify}}$: used only for model selection scoring and ranking.
\item \textbf{Test/forecast block} $\mathcal{T}_{\text{test}}$: used once for final reporting after selection.
\end{itemize}

\subsection{Rolling-origin evaluation}
Let $\mathcal{O}$ be the set of forecast origins within $\mathcal{T}_{\text{verify}}$ and $\mathcal{H}$ the horizon set. We evaluate candidates at pairs $(t_0,h)$ such that $y_{t_0+h}$ exists within the verification window. Weights $w_{t_0,h}\ge 0$ are deterministic and sum to 1.

\subsection{Quantile grid and synthesis}
Let $\mathcal{Q}=\{\tau_1,\dots,\tau_M\}$ be the quantile set (\texttt{p\_vec} in config). Synthesis is handled by
\texttt{exdqlm\_synthesize\_from\_draws()} with isotonic and rearrangement options controlled by \texttt{cfg$synthesis}.

\section{Candidate DESN search space (reproducible defaults)}

\subsection{Hard constraints}
\begin{itemize}[itemsep=2pt, topsep=4pt]
\item \texttt{length(n) == D}
\item \texttt{length(alpha) == D} or scalar-expand across layers
\item \texttt{length(rho) == D} or scalar-expand across layers
\item \texttt{length(n\_tilde) == D-1} when $D>1$ (empty when $D=1$)
\item \texttt{pi\_w, pi\_in, washout, add\_bias} fixed unless explicitly varied
\end{itemize}

\subsection{Default coarse ranges (override by config)}
\begin{itemize}[itemsep=2pt, topsep=4pt]
\item $D \in \{1,2,3\}$
\item $n_1 \in \{300,500,800,1000,1500\}$ and layer ratios $r_2,r_3 \in (0,1]$ (if using ratios)
\item $m \in \{30,60,90,120,180\}$
\item $\alpha \in [0.05, 0.4]$
\item $\rho \in [0.85, 0.99]$
\end{itemize}
Default candidate generation uses $n_1$ plus ratios $(r_2,r_3)$, then expands to the full $n$ vector. Direct specification of the full $n$ vector is also allowed.

\section{Calibration metric redesign (CalCRPS only)}
\label{sec:calcrps}
Calibration is defined \textbf{only} via a CalCRPS metric computed from posterior/variational \emph{coverage draws}.

Let $E$ be the evaluation lattice of $(t_0,h)$ in the verification window, with deterministic weights $w_{t_0,h} \ge 0$ that sum to 1. For each $\tau \in \mathcal{Q}$ and each posterior/variational draw $r=1..R$, define weighted empirical coverage:
\[
 c_{\tau}^{(r)} = \sum_{(t_0,h)\in E} w_{t_0,h} \cdot \Ind\{ y_{t_0+h} \le q_{t_0,h,\tau}^{(r)} \}.
\]
This yields draws $\{c_{\tau}^{(r)}\}_{r=1}^R$ in $[0,1]$. Let $\widehat F_{c_\tau}$ be the empirical CDF of these draws. Define:
\[
\text{CalCRPS}_\tau = \CRPS\big(\widehat F_{c_\tau},\, \text{target}=\tau\big).
\]

\paragraph{Semantics of draw $r$.} Ideally, each $r$ is a \emph{joint draw} that induces \(q_{t_0,h,\tau}^{(r)}\) for all $(t_0,h)\in E$ under shared parameters/latents. If only per-time draws are available, approximate by treating each draw as conditionally independent across $(t_0,h)$ and note this as an approximation.

\paragraph{Empirical CRPS for coverage draws (optional).} For draws \(\{c^{(r)}\}_{r=1}^R\) and target \(x\),
\[
\CRPS(\widehat F, x) = \frac{1}{R}\sum_{r=1}^R |c^{(r)}-x| - \frac{1}{2R^2}\sum_{r=1}^R\sum_{r'=1}^R |c^{(r)}-c^{(r')}|.
\]
Implementation note: compute this in \(O(R\log R)\) using sorting-based formulas; avoid \(O(R^2)\) double sums for large \(R\). A standard approach uses sorted samples and cumulative sums.

Aggregate across $\tau$:
\[
\text{CalCRPS}_{\text{mean}} = \frac{1}{|\mathcal{Q}|}\sum_{\tau\in\mathcal{Q}} \text{CalCRPS}_\tau,
\quad
\text{CalCRPS}_{\text{max}} = \max_{\tau\in\mathcal{Q}} \text{CalCRPS}_\tau.
\]
Optional extension: compute CalCRPS per lead group (e.g., short/medium/long or per $h$) and aggregate; this can diagnose lead-specific miscalibration without changing the primary objective.

\section{Selection objective (CalCRPS only)}
\paragraph{Primary:} $\CRPS_{\text{synth}}$ of the synthesized predictive distribution (verification window).
\paragraph{Secondary:} CalCRPS (defined above).

\subsection*{Option A: constrained}
\[
\min_\theta\; \CRPS_{\text{synth}}(\theta)
\quad \text{subject to}\quad
\text{CalCRPS}_{\text{mean}}(\theta) \le \varepsilon
\]
(Optionally constrain \text{CalCRPS}_{\text{max}} instead of mean.)

\subsection*{Option B: weighted sum}
\[
J_\lambda(\theta) = \CRPS_{\text{synth}}(\theta) + \lambda\,\text{CalCRPS}_{\text{mean}}(\theta)
\]
(Optionally replace mean with max.)

\paragraph{Repo note.} Current repo code primarily selects by \texttt{CRPS\_synth}; CalCRPS is a planned v2 extension.

\item \texttt{verify\_window\_id}: hash of \((t_{\text{verify start}}, t_{\text{verify end}}, \text{lead\_set\_id}, \text{origins\_spec\_id}, \text{weight\_spec\_id})\) or store explicit start/end and use a hash as id.

\section{Multi-fidelity stages (planned defaults)}
Define stages in terms of seeds, quantile grid size, horizons, origins, and draws $R$.

\begin{center}
\begin{tabular}{@{}lccccc@{}}
\toprule
Stage & Seeds $S$ & $|\mathcal{Q}|$ & $|\mathcal{H}|$ / leads & Origins policy & Draws $R$ \\
\midrule
Coarse & 1 & 3 (e.g., 0.1/0.5/0.9) & small set (e.g., 1,7,14) & subsample (e.g., 25\%) & 500--1000 \\
Final  & 5--10 & full (cfg $p\_vec$) & full set (cfg eval\_leads) & all origins & 2000--5000 \\
\bottomrule
\end{tabular}
\end{center}
\noindent\textbf{Origin subsampling rule (required):} subsampling must be deterministic given a fixed seed (or a stride rule), and the same subsampled origin set must be used for all candidates within a stage.
